# Tooltip

This is a wrapper that adds a tooltip to a child element.
 
## Properties
<Props of={Tooltip} />

## Basic usage

<Playground>
    <Tooltip
        renderTrigger={() => <div>some normal text with a tooltip</div>}
        renderContent={'and some normal text in the tooltip'}
        position='top'
    />
    <Tooltip
        renderTrigger={<Button fullWidth>a demo tooltip with more complex content and trigger</Button>}
        renderContent={<Image src='https://lesbaa.github.io/img/z-loki.gif' text='an image!' />}
        position='right'
    />
    <Tooltip
        renderTrigger={<Button>a demo tooltip with a function render prop</Button>}
        renderContent={() => <Image src='https://lesbaa.github.io/img/z-loki.gif' text='an image but in a functional render prop!' />}
        position='right'
    />
    <Tooltip
        renderTrigger={<Button>a demo tooltip with the tooltip disabled</Button>}
        renderContent={() => <Image src='https://lesbaa.github.io/img/z-loki.gif' text='an image but in a functional render prop!' />}
        position='right'
        disableTooltip
    />
    <Tooltip
        renderTrigger={<Button>a demo tooltip with invert == true</Button>}
        renderContent={() => <Image src='https://lesbaa.github.io/img/z-loki.gif' text='an image but in an inverted tooltip!' />}
        position='right'
        invert
    />
</Playground>
