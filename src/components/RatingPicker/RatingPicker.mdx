import { useState } from 'react'
import RatingPicker from './'
import { Playground, Props } from 'docz'

# RatingPicker

This is the ratings toggle component used inside the note form modals.
 
## Properties
<Props of={RatingPicker} />

## Basic usage
<Playground>
{() => {
    const [rating, setRating] = useState(null)
    const [rating2, setRating2] = useState(4)
    return (
      <div>
      <p>
        Standard Picker
        <RatingPicker inputName="rating1" rating={rating} onChange={value => setRating(value)} />
      </p>
      <p>
        Standard Picker
        <RatingPicker inputName="rating2" rating={rating2} onChange={value => setRating2(value)} />
      </p>
      <p>
        Static Picker
        <RatingPicker inputName="rating2" isStatic rating={3} />
      </p>
      </div>
    )
}}
</Playground>
