## Basic usage

### With Dropdown
<Playground>
{() => {
    const [value, setValue] = useState('')
    const options = [1,2,3].map(num => ({
        name: String(num),
        id: String(num),
        value: String(num)
    }))
    const [value2, setValue2] = useState(options)
    const [nullValue, setNullValue] = useState(null)
  
    return (
        <Fragment>
         <BlockContainer style={{
             marginBottom: '25px'
         }}>
             <div>Type 1, 2, or 3 to filter options.</div>
             <Search 
                options={value2}
                value={value}
                name='search'
                onInputChange={e => { 
                    const inputVal = e.target.value
                    setValue(inputVal)
                    if (inputVal) {
                        setValue2(options.filter(num => num.value === inputVal))
                    } else {
                        setValue2(options)
                    }
                }}
                onSelectChange={e => {
                    setValue(e.target.value)
                    setValue2(options)
                }}
                hasDropdown
             />
         </BlockContainer>
        </Fragment>
    )    
}}
</Playground>
